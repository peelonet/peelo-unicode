36  inline void encode_codepoint(char32_t c, std::string& output)
37  {
38  if (c > 0xffff)
39  {
40  const auto high = (c >> 10) + 0xd7c0;
41  const auto low = (c & 0x3ff) + 0xdc00;
42 
43  output.append(1, static_cast<unsigned char>(high & 0xff));
44  output.append(1, static_cast<unsigned char>((high >> 8) & 0xff));
45  output.append(1, static_cast<unsigned char>(low & 0xff));
46  output.append(1, static_cast<unsigned char>((low >> 8) & 0xff));
47  } else {
48  output.append(1, static_cast<unsigned char>(c & 0xff));
49  output.append(1, static_cast<unsigned char>((c & 0xff00) >> 8));
50  }
51  }

87  inline bool decode_advance(
88  const char* input,
89  std::string::size_type& i,
90  const std::string::size_type length,
91  char32_t& output
92  )
93  {
94  unsigned char p0;
95  unsigned char p1;
96  unsigned char p2;
97  unsigned char p3;
98 
99  if (i + 1 < length && (input[i + 1] & 0xfc) == 0xd8)
100  {
101  if (i + 3 >= length)
102  {
103  return false;
104  }
105  p0 = static_cast<unsigned char>(input[i]);
106  p1 = static_cast<unsigned char>(input[i + 1]);
107  p2 = static_cast<unsigned char>(input[i + 2]);
108  p3 = static_cast<unsigned char>(input[i + 3]);
109  output = static_cast<char32_t>(
110  ((((p1 << 8) + p0) & 0x03ff) << 10)
111  + (((p3 << 8) + p2) & 0x03ff) + 0x10000
112  );
113  i += 4;
114  } else {
115  if (i + 1 >= length)
116  {
117  return false;
118  }
119  p0 = static_cast<unsigned char>(input[i]);
120  p1 = static_cast<unsigned char>(input[i + 1]);
121  output = static_cast<char32_t>(p1 * 256 + p0);
122  i += 2;
123  }
124 
125  return true;
126  }
