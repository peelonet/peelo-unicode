42  template<class CharT = char32_t, class ByteT = char>
43  static void encode_codepoint(CharT c, std::basic_string<ByteT>& output)
44  {
45  if (c > 0xffff)
46  {
47  const auto plane = (c >> 16) - 1;
48  const auto high = (c & 0xff00) >> 8;
49 
50  output.append(1, static_cast<ByteT>((plane >> 2) & 0xd8));
51  output.append(1, static_cast<ByteT>(((plane & 0x03) << 6) + (high >> 2)));
52  output.append(1, static_cast<ByteT>((high & 0x03) + 0xdc));
53  output.append(1, static_cast<ByteT>(c & 0xff));
54  } else {
55  output.append(1, static_cast<ByteT>((c & 0xff00) >> 8));
56  output.append(1, static_cast<ByteT>(c & 0xff));
57  }
58  }
59  }
